Introduction

Internal migration matters because population movement changes state populations, and state populations determine political power through congressional apportionment. In this report, I use Census ACS migration flow data to study movement between states and metro areas. I then use recent migration patterns to produce a simple forecast of 2030 state populations and estimate how congressional seats could be distributed after the next Census.

  purrr::map_dfr(years, function(y){
    get_acs(
      geography = "state",
      variables = c(total_population = "B01003_001"),
      year = y,
      survey = "acs1",
      geometry = TRUE,

      st_drop_geometry() |>
      transmute(
        state = NAME,
        state_abbr = state.abb[match(NAME, state.name)],
        year = y,
        total_population = estimate
      )
  }) |>
    filter(!is.na(state_abbr))
}

clean_migration_value <- function(x){
  x |>
    as.character() |>
    str_replace_all(",", "") |>
    str_replace_all("^N$", NA_character_) |>
    str_replace_all("^X$", NA_character_) |>

      year_current = 2023L,

estimate_one_year_parameters <- function(flow_data, current_year){
  pop_current <- pop_for_model |>
    filter(year == current_year) |>
    select(state_current = state_abbr, pop_current = total_population)
  
  pop_prior <- pop_for_model |>
    filter(year == current_year - 1) |>
    select(state_1y = state_abbr, pop_1y = total_population)
  
  national_current <- sum(pop_current$pop_current, na.rm = TRUE)
  national_prior <- sum(pop_prior$pop_1y, na.rm = TRUE)
  
  domestic_flows <- flow_data |>
    filter(year_current == current_year) |>
    filter(state_current != state_1y)
  
  natural_growth_rate <- (national_current - national_prior) / national_prior
  
  migration_rates <- domestic_flows |>
    left_join(pop_current, by = "state_current") |>
    left_join(pop_prior, by = "state_1y") |>
    mutate(
      migration_rate = population / (pop_current * pop_1y)
    ) |>
    select(state_current, state_1y, migration_rate)
  
  list(
    natural_growth_rate = natural_growth_rate,
    migration_rates = migration_rates
  )
}

project_one_year <- function(pop_table, from_year, to_year, natural_growth_rate, migration_rates){
  pop_from <- pop_table |>
    filter(year == from_year) |>
    select(state_abbr, population = projected_population)
  
  out_effects <- migration_rates |>
    left_join(pop_from, by = c("state_1y" = "state_abbr")) |>
    group_by(state_1y) |>
    summarize(
      projected_out = sum(migration_rate * population, na.rm = TRUE),
      .groups = "drop"
    ) |>
    rename(state_abbr = state_1y)
  
  in_effects <- migration_rates |>
    left_join(pop_from, by = c("state_1y" = "state_abbr")) |>
    group_by(state_current) |>
    summarize(
      projected_in = sum(migration_rate * population, na.rm = TRUE),
      .groups = "drop"
    ) |>
    rename(state_abbr = state_current)
  
  pop_from |>
    left_join(in_effects, by = "state_abbr") |>
    left_join(out_effects, by = "state_abbr") |>
    mutate(
      projected_in = replace_na(projected_in, 0),
      projected_out = replace_na(projected_out, 0),
      projected_population = population * (1 + natural_growth_rate) +
        projected_in - projected_out,
      year = to_year
    ) |>
    select(state_abbr, year, projected_population)
}

huntington_hill <- function(pop_data, max_districts = 100, n_districts = 435){
  pop_data |>
    select(state, state_abbr, population = projected_population) |>
    mutate(population = as.numeric(population)) |>
    expand_grid(cd = seq(1, max_districts)) |>
    mutate(
      hh_den = sqrt(cd * (cd + 1)),
      hh_weight = population / hh_den
    ) |>
    slice_max(hh_weight, n = n_districts - 50, with_ties = FALSE) |>
    count(state, state_abbr, population, name = "additional_seats") |>
    mutate(
      n_districts = additional_seats + 1,
      pop_per_district = population / n_districts
    ) |>
    arrange(desc(n_districts), desc(population))
}
