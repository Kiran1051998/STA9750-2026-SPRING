Introduction

Internal migration matters because population movement changes state populations, and state populations determine political power through congressional apportionment. In this report, I use Census ACS migration flow data to study movement between states and metro areas. I then use recent migration patterns to produce a simple forecast of 2030 state populations and estimate how congressional seats could be distributed after the next Census.

This report finds that internal migration is not evenly distributed across the United States. Recent population growth is concentrated in faster-growing Sun Belt and Mountain West states, while New York faces pressure from out-migration to other large states and nearby metro areas. For New York, migration appears to be a political representation challenge because sustained population losses can reduce its relative share of the national population. The projection model used here is intentionally simple, but it helps show how even modest annual migration patterns can matter when congressional seats are allocated after the next Census.

get_state_population <- function(years = c(2015:2019, 2021:2024)){
  purrr::map_dfr(years, function(y){
    get_acs(
      geography = "state",
      variables = c(total_population = "B01003_001"),
      year = y,
      survey = "acs1",
      geometry = TRUE,
      cache_table = TRUE
    ) |>
      st_drop_geometry() |>
      transmute(
        state = NAME,
        state_abbr = state.abb[match(NAME, state.name)],
        year = y,
        total_population = estimate
      )
  }) |>
    filter(!is.na(state_abbr))
}

State-to-State Migration Flows

The state-to-state migration data comes from Census Excel files. The 2024 file is in a long format, while the 2023 file is in a wide matrix format. Because of this, the two years require different cleaning approaches before they can be combined.

clean_migration_value <- function(x){
  x |>
    as.character() |>
    str_replace_all(",", "") |>
    str_replace_all("^N$", NA_character_) |>
    str_replace_all("^X$", NA_character_) |>
    str_replace_all("^N/A$", NA_character_) |>
    as.integer()
}

state_to_state_migration_2024 <- function(){

  flows <- read_excel(
    file_2024,
    sheet = "Table",
    skip = 5,
    col_names = c("state_current", "state_1y", "population", "moe")
  ) |>
    filter(!is.na(state_current)) |>
    filter(
      state_current %in% state.name,
      state_1y %in% state.name
    ) |>
    filter(state_current != state_1y) |>
    filter(population != "Estimate") |>
    transmute(
      state_current = state.abb[match(state_current, state.name)],
      state_1y = state.abb[match(state_1y, state.name)],
      population = clean_migration_value(population),
      year_current = 2024L,
      year_1y = 2023L
    ) |>
    filter(!is.na(population))

  stayed <- read_excel(
    file_2024,
    sheet = "Supplemental - Current Res",
    skip = 5,
    col_names = FALSE
  ) |>
    filter(...1 %in% state.name) |>
    transmute(
      state_current = state.abb[match(...1, state.name)],
      state_1y = state_current,
      population = clean_migration_value(...6),
      year_current = 2024L,
      year_1y = 2023L
    )

  bind_rows(flows, stayed)
}

2023 State-to-State Migration

state_to_state_migration_2023 <- function(){

  raw <- read_excel(
    file_2023,
    sheet = "Table",
    skip = 7,
    col_names = FALSE
  )

  data_rows <- raw |>
    filter(...1 %in% state.name)

  stayed <- data_rows |>
    transmute(
      state_current = state.abb[match(...1, state.name)],
      state_1y = state_current,
      population = clean_migration_value(...6),
      year_current = 2023L,
      year_1y = 2022L
    )

  estimate_cols <- seq(10, 110, by = 2)

  state_lookup <- c(
    state.name[1:8],
    "District of Columbia",
    state.name[9:50]
  )

  names(data_rows)[estimate_cols] <- state_lookup

  flows <- data_rows |>
    select(state_current_name = ...1, all_of(state_lookup)) |>
    pivot_longer(
      cols = all_of(state_lookup),
      names_to = "state_1y_name",
      values_to = "population"
    ) |>
    filter(state_current_name != state_1y_name) |>
    transmute(
      state_current = state.abb[match(state_current_name, state.name)],
      state_1y = state.abb[match(state_1y_name, state.name)],
      population = clean_migration_value(population),
      year_current = 2023L,
      year_1y = 2022L
    ) |>
    filter(!is.na(population))

  bind_rows(flows, stayed)
}

Metro-to-Metro Migration Flows

Code 
pull_state_from_metro <- function(metro_name){
  str_extract(metro_name, ".*, (\\S{2})[-[:alpha:]]* Metro Area", group = 1)
}

metro_to_metro_migration <- function(){
  data_file <- fs::path(data_mp03, "metro_to_metro_migration_2016_2020.json")
  
  if(!fs::file_exists(data_file)){
    metro_url <- paste0(
      "https://api.census.gov/data/2020/acs/flows?get=",
      "MOVEDIN,MOVEDOUT,FULL1_NAME,FULL2_NAME",
      "&for=metropolitan%20statistical%20area/micropolitan%20statistical%20area:*"
    )
    
    download.file(
      url = metro_url,
      destfile = data_file,
      mode = "wb"
    )
  }
  
  raw_json <- jsonlite::fromJSON(data_file)
  
  metro_df <- as_tibble(
    raw_json[-1, ],
    .name_repair = "minimal"
  )
  
  names(metro_df) <- raw_json[1, ]
  
  metro_df |>
    janitor::clean_names() |>
    transmute(
      metro_current = full1_name,
      metro_1y = full2_name,
      moved_in = as.integer(movedin),
      moved_out = as.integer(movedout),
      metro_current_state = pull_state_from_metro(full1_name),
      metro_1y_state = pull_state_from_metro(full2_name)
    )
}

state_migration_summary <- state_migration |>
  filter(year_current == latest_state_year) |>
  filter(state_current != state_1y) |>
  group_by(state_current) |>
  summarize(in_migration = sum(population, na.rm = TRUE), .groups = "drop") |>
  full_join(
    state_migration |>
      filter(year_current == latest_state_year) |>
      filter(state_current != state_1y) |>
      group_by(state_1y) |>
      summarize(out_migration = sum(population, na.rm = TRUE), .groups = "drop"),
    by = c("state_current" = "state_1y")
  ) |>
  rename(state_abbr = state_current) |>
  left_join(states_df, by = "state_abbr") |>
  mutate(
    in_migration = replace_na(in_migration, 0),
    out_migration = replace_na(out_migration, 0),
    net_migration = in_migration - out_migration
  ) |>
  arrange(desc(net_migration))

estimate_one_year_parameters <- function(flow_data, current_year){
  pop_current <- pop_for_model |>
    filter(year == current_year) |>
    select(state_current = state_abbr, pop_current = total_population)
  
  pop_prior <- pop_for_model |>
    filter(year == current_year - 1) |>
    select(state_1y = state_abbr, pop_1y = total_population)
  
  national_current <- sum(pop_current$pop_current, na.rm = TRUE)
  national_prior <- sum(pop_prior$pop_1y, na.rm = TRUE)
  
  domestic_flows <- flow_data |>
    filter(year_current == current_year) |>
    filter(state_current != state_1y)
  
  natural_growth_rate <- (national_current - national_prior) / national_prior
  
  migration_rates <- domestic_flows |>
    left_join(pop_current, by = "state_current") |>
    left_join(pop_prior, by = "state_1y") |>
    mutate(
      migration_rate = population / (pop_current * pop_1y)
    ) |>
    select(state_current, state_1y, migration_rate)
  
  list(
    natural_growth_rate = natural_growth_rate,
    migration_rates = migration_rates
  )
}

The model averages the 2023 and 2024 national growth rates and the observed state-to-state migration rates. This keeps the forecast simple and transparent, but it also means the projection should be interpreted as a rough scenario rather than a precise demographic forecast.

Code 
project_one_year <- function(pop_table, from_year, to_year, natural_growth_rate, migration_rates){
  pop_from <- pop_table |>
    filter(year == from_year) |>
    select(state_abbr, population = projected_population)
  
  out_effects <- migration_rates |>
    left_join(pop_from, by = c("state_1y" = "state_abbr")) |>
    group_by(state_1y) |>
    summarize(
      projected_out = sum(migration_rate * population, na.rm = TRUE),
      .groups = "drop"
    ) |>
    rename(state_abbr = state_1y)
  
  in_effects <- migration_rates |>
    left_join(pop_from, by = c("state_1y" = "state_abbr")) |>
    group_by(state_current) |>
    summarize(
      projected_in = sum(migration_rate * population, na.rm = TRUE),
      .groups = "drop"
    ) |>
    rename(state_abbr = state_current)
  
  pop_from |>
    left_join(in_effects, by = "state_abbr") |>
    left_join(out_effects, by = "state_abbr") |>
    mutate(
      projected_in = replace_na(projected_in, 0),
      projected_out = replace_na(projected_out, 0),
      projected_population = population * (1 + natural_growth_rate) +
        projected_in - projected_out,
      year = to_year
    ) |>
    select(state_abbr, year, projected_population)
}

huntington_hill <- function(pop_data, max_districts = 100, n_districts = 435){
  pop_data |>
    select(state, state_abbr, population = projected_population) |>
    mutate(population = as.numeric(population)) |>
    expand_grid(cd = seq(1, max_districts)) |>
    mutate(
      hh_den = sqrt(cd * (cd + 1)),
      hh_weight = population / hh_den
    ) |>
    slice_max(hh_weight, n = n_districts - 50, with_ties = FALSE) |>
    count(state, state_abbr, population, name = "additional_seats") |>
    mutate(
      n_districts = additional_seats + 1,
      pop_per_district = population / n_districts
    ) |>
    arrange(desc(n_districts), desc(population))
}

How Much Migration Is Needed for One More Seat?

Code 
estimate_extra_people_for_next_seat <- function(apportionment_table, focus_state_abbr){
  base_pop <- apportionment_table |>
    select(state, state_abbr, projected_population = population)
  
  baseline_seats <- apportionment_table |>
    filter(state_abbr == focus_state_abbr) |>
    pull(n_districts)
  
  increments <- seq(0, 1000000, by = 10000)
  
  results <- purrr::map_dfr(increments, function(extra_people){
    adjusted <- base_pop |>
      mutate(
        projected_population = if_else(
          state_abbr == focus_state_abbr,
          projected_population + extra_people,
          projected_population
        )
      )
    
    seats <- huntington_hill(adjusted) |>
      filter(state_abbr == focus_state_abbr) |>
      pull(n_districts)
    
    tibble(extra_people = extra_people, seats = seats)
  })
  
  results |>
    filter(seats > baseline_seats) |>
    slice_head(n = 1)
}

This project shows how migration patterns can shape both population growth and political representation. States that attract more residents from other states are more likely to gain congressional power, while states losing residents risk losing seats. For New York, the key policy challenge is to understand where movers are coming from, where residents are going, and how to convert those existing migration pathways into long-term population growth.

The results show that internal migration is a major force shaping state population change and future congressional representation. Fast-growing states are concentrated in the Sun Belt and Mountain West, while New York faces pressure from residents leaving for other states and metro areas. New York’s strongest migration relationships appear in both the state-level and metro-level tables, especially through the New York City metro area.
